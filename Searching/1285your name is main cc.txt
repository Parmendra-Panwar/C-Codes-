#include <bits/stdc++.h>
using namespace std;

bool  issubseq(string A, string B){
  int i=0,j=0;
  
  while(i<A.length() && j<B.length()){
    if(A[i]==B[j]){
      i++;
    }
    j++;
  }
  return(i==A.length());
}

int main() {
	int t;
	cin>>t;
	while(t--){
	   string A,B;
	   cin>>A>>B;
	   if(issubseq(A,B) or issubseq(B,A) == 1){
	     cout<<"YES"<<endl;
	   }else{
	     cout<<"NO"<<endl;
	   }
	}
}
