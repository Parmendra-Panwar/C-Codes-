#include <bits/stdc++.h>
using namespace std;

#define int long long

void solve() {
    int n;
    cin >> n;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; ++i)
        cin >> a[i];
    int ans = 0;
    vector<int> powers(32); 
    powers[0] = 1;
    for (int i = 1; i < 32; ++i)
        powers[i] = powers[i - 1] * a[1]; 
    for (int i = 1; i <= n; ++i) {
        int aitoj = a[i];
        if (aitoj == 1) {
            ans += n;
        } else {
            for (int j = 1; j <= min(n, 30LL); ++j) { 
                if (aitoj <= a[j])
                    ans++;
                aitoj *= a[i];
                if (aitoj > 1e9)
                    break;
            }
        }
    }
    cout << ans << endl;
}

int32_t main() {
#ifndef ONLINE_JUDGE
    freopen("Input.txt", "r", stdin);
    freopen("Output.txt", "w", stdout);
#endif
    int t = 1;
    cin >> t;
    while (t--)
        solve();

    return 0;
}
